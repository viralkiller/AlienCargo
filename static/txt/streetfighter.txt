    <script>
        // Core configuration
        const GAME_WIDTH = 800;
        const GAME_HEIGHT = 400;
        const GRAVITY = 1200;
        const SPEED = 250;
        const JUMP_VELOCITY = -600;
        const MAX_HEALTH = 100;

        class FightScene extends Phaser.Scene {
            constructor() {
                super('FightScene');
                this.p1State = { health: MAX_HEALTH, isAttacking: false, isDead: false };
                this.p2State = { health: MAX_HEALTH, isAttacking: false, isDead: false };
            }

            preload() {
                console.log("[SYSTEM] Preload started. Generating dynamic textures.");

                // Generate P1 Texture (Red Box)
                let g = this.add.graphics();
                g.fillStyle(0xcc0000); g.fillRect(0, 0, 50, 100);
                g.generateTexture('p1', 50, 100);
                g.clear();

                // Generate P2 Texture (Blue Box)
                g.fillStyle(0x0066cc); g.fillRect(0, 0, 50, 100);
                g.generateTexture('p2', 50, 100);
                g.clear();

                // Generate Attack Hitbox (Yellow)
                g.fillStyle(0xffff00, 0.5); g.fillRect(0, 0, 60, 20);
                g.generateTexture('hitbox', 60, 20);
                g.destroy();

                console.log("[SYSTEM] Preload complete.");
            }

            create() {
                console.log("[SYSTEM] Create started. Initializing entities and environment.");

                // Environment setup
                this.physics.world.setBounds(0, 0, GAME_WIDTH, GAME_HEIGHT);

                // Ground
                this.ground = this.add.rectangle(GAME_WIDTH/2, GAME_HEIGHT - 10, GAME_WIDTH, 20, 0x555555);
                this.physics.add.existing(this.ground, true);

                // Player 1 Setup
                this.p1 = this.physics.add.sprite(200, GAME_HEIGHT - 100, 'p1');
                this.p1.setCollideWorldBounds(true);
                this.physics.add.collider(this.p1, this.ground);

                this.p1Hitbox = this.physics.add.sprite(this.p1.x, this.p1.y, 'hitbox');
                this.p1Hitbox.setVisible(false);
                this.p1Hitbox.body.setAllowGravity(false);

                // Player 2 Setup
                this.p2 = this.physics.add.sprite(600, GAME_HEIGHT - 100, 'p2');
                this.p2.setCollideWorldBounds(true);
                this.physics.add.collider(this.p2, this.ground);

                this.p2Hitbox = this.physics.add.sprite(this.p2.x, this.p2.y, 'hitbox');
                this.p2Hitbox.setVisible(false);
                this.p2Hitbox.body.setAllowGravity(false);

                // Prevent players from walking through each other
                this.physics.add.collider(this.p1, this.p2);

                // Inputs
                this.keys = this.input.keyboard.addKeys({
                    p1Up: Phaser.Input.Keyboard.KeyCodes.W,
                    p1Left: Phaser.Input.Keyboard.KeyCodes.A,
                    p1Right: Phaser.Input.Keyboard.KeyCodes.D,
                    p1Attack: Phaser.Input.Keyboard.KeyCodes.F,
                    p2Up: Phaser.Input.Keyboard.KeyCodes.UP,
                    p2Left: Phaser.Input.Keyboard.KeyCodes.LEFT,
                    p2Right: Phaser.Input.Keyboard.KeyCodes.RIGHT,
                    p2Attack: Phaser.Input.Keyboard.KeyCodes.SPACE
                });

                // Input Logging Hooks
                this.setupInputLogging();

                console.log("[SYSTEM] Game objects instantiated. Awaiting interaction.");
            }

            setupInputLogging() {
                // Detailed logging for UI/Controls to track sequence of events
                this.input.keyboard.on('keydown', (event) => {
                    const keyMap = { 87: 'W (P1 Up)', 65: 'A (P1 Left)', 68: 'D (P1 Right)', 70: 'F (P1 Attack)', 38: 'UP (P2 Up)', 37: 'LEFT (P2 Left)', 39: 'RIGHT (P2 Right)', 32: 'SPACE (P2 Attack)' };
                    if (keyMap[event.keyCode]) {
                        console.log(`[UI INTERACTION] KeyDown detected: ${keyMap[event.keyCode]}`);
                    }
                });

                this.input.keyboard.on('keyup', (event) => {
                    const keyMap = { 87: 'W (P1 Up)', 65: 'A (P1 Left)', 68: 'D (P1 Right)', 70: 'F (P1 Attack)', 38: 'UP (P2 Up)', 37: 'LEFT (P2 Left)', 39: 'RIGHT (P2 Right)', 32: 'SPACE (P2 Attack)' };
                    if (keyMap[event.keyCode]) {
                        console.log(`[UI INTERACTION] KeyUp detected: ${keyMap[event.keyCode]}`);
                    }
                });
            }

            update() {
                if (this.p1State.isDead || this.p2State.isDead) return;

                this.handlePlayer1();
                this.handlePlayer2();
                this.updateHitboxes();
            }

            handlePlayer1() {
                // Movement
                if (this.keys.p1Left.isDown) {
                    this.p1.setVelocityX(-SPEED);
                } else if (this.keys.p1Right.isDown) {
                    this.p1.setVelocityX(SPEED);
                } else {
                    this.p1.setVelocityX(0);
                }

                // Jump
                if (this.keys.p1Up.isDown && this.p1.body.touching.down) {
                    console.log("[ACTION] P1 Executed Jump");
                    this.p1.setVelocityY(JUMP_VELOCITY);
                }

                // Attack
                if (Phaser.Input.Keyboard.JustDown(this.keys.p1Attack) && !this.p1State.isAttacking) {
                    console.log("[ACTION] P1 Executed Attack");
                    this.triggerAttack('p1');
                }
            }

            handlePlayer2() {
                // Movement
                if (this.keys.p2Left.isDown) {
                    this.p2.setVelocityX(-SPEED);
                } else if (this.keys.p2Right.isDown) {
                    this.p2.setVelocityX(SPEED);
                } else {
                    this.p2.setVelocityX(0);
                }

                // Jump
                if (this.keys.p2Up.isDown && this.p2.body.touching.down) {
                    console.log("[ACTION] P2 Executed Jump");
                    this.p2.setVelocityY(JUMP_VELOCITY);
                }

                // Attack
                if (Phaser.Input.Keyboard.JustDown(this.keys.p2Attack) && !this.p2State.isAttacking) {
                    console.log("[ACTION] P2 Executed Attack");
                    this.triggerAttack('p2');
                }
            }

            updateHitboxes() {
                // Keep hitboxes pinned to players, adjust based on relative position to face each other
                const p1FacingRight = this.p1.x < this.p2.x;
                const p2FacingLeft = this.p2.x > this.p1.x;

                this.p1Hitbox.x = p1FacingRight ? this.p1.x + 40 : this.p1.x - 40;
                this.p1Hitbox.y = this.p1.y - 10;

                this.p2Hitbox.x = p2FacingLeft ? this.p2.x - 40 : this.p2.x + 40;
                this.p2Hitbox.y = this.p2.y - 10;
            }

            triggerAttack(playerKey) {
                const isP1 = playerKey === 'p1';
                const state = isP1 ? this.p1State : this.p2State;
                const hitbox = isP1 ? this.p1Hitbox : this.p2Hitbox;
                const target = isP1 ? this.p2 : this.p1;

                state.isAttacking = true;
                hitbox.setVisible(true);

                // Frame window for hit detection
                let hitDetected = false;

                // Simple overlap check
                this.time.addEvent({
                    delay: 50,
                    callback: () => {
                        if (this.physics.overlap(hitbox, target)) {
                            console.log(`[COLLISION] ${playerKey.toUpperCase()} Attack Connected!`);
                            this.applyDamage(isP1 ? 'p2' : 'p1', 10);
                            hitDetected = true;
                        } else {
                            console.log(`[COLLISION] ${playerKey.toUpperCase()} Attack Missed.`);
                        }
                    }
                });

                // End Attack Frame
                this.time.addEvent({
                    delay: 200,
                    callback: () => {
                        state.isAttacking = false;
                        hitbox.setVisible(false);
                    }
                });
            }

            applyDamage(targetKey, amount) {
                const targetState = targetKey === 'p1' ? this.p1State : this.p2State;
                const targetSprite = targetKey === 'p1' ? this.p1 : this.p2;

                targetState.health -= amount;
                if (targetState.health < 0) targetState.health = 0;

                console.log(`[STATE UPDATE] ${targetKey.toUpperCase()} took ${amount} damage. Current Health: ${targetState.health}`);

                // Update DOM UI
                const fillElement = document.getElementById(`${targetKey}-health-fill`);
                if (fillElement) {
                    fillElement.style.width = `${(targetState.health / MAX_HEALTH) * 100}%`;
                }

                // Flash damage color
                targetSprite.setTint(0xffffff);
                this.time.delayedCall(100, () => targetSprite.clearTint());

                // Knockback
                const knockbackDir = targetKey === 'p1' ? (this.p1.x < this.p2.x ? -1 : 1) : (this.p2.x < this.p1.x ? -1 : 1);
                targetSprite.setVelocityX(200 * knockbackDir);
                targetSprite.setVelocityY(-150);

                if (targetState.health <= 0) {
                    this.handleKO(targetKey);
                }
            }

            handleKO(loserKey) {
                console.log(`[GAME EVENT] KO! ${loserKey.toUpperCase()} has been defeated.`);
                if (loserKey === 'p1') this.p1State.isDead = true;
                if (loserKey === 'p2') this.p2State.isDead = true;

                const loserSprite = loserKey === 'p1' ? this.p1 : this.p2;
                loserSprite.setTint(0x444444);
                loserSprite.angle = 90; // lay flat

                const winText = loserKey === 'p1' ? 'PLAYER 2 WINS' : 'PLAYER 1 WINS';

                const t = this.add.text(GAME_WIDTH/2, GAME_HEIGHT/2, winText, {
                    fontSize: '48px',
                    fontFamily: 'monospace',
                    color: '#fff',
                    stroke: '#000',
                    strokeThickness: 6
                }).setOrigin(0.5);

                console.log(`[UI UPDATE] Displaying win text: ${winText}`);

                // Show restart button
                const resetBtn = document.getElementById('reset-btn');
                resetBtn.style.display = 'block';

                resetBtn.onclick = () => {
                    console.log("[UI INTERACTION] Restart Button Clicked.");
                    resetBtn.style.display = 'none';
                    // Reset HTML UI
                    document.getElementById('p1-health-fill').style.width = '100%';
                    document.getElementById('p2-health-fill').style.width = '100%';
                    console.log("[SYSTEM] Restarting Scene...");
                    this.scene.restart();
                };
            }
        }

        const config = {
            type: Phaser.AUTO,
            parent: 'game-container',
            width: GAME_WIDTH,
            height: GAME_HEIGHT,
            physics: {
                default: 'arcade',
                arcade: {
                    gravity: { y: GRAVITY },
                    debug: false
                }
            },
            scene: FightScene
        };

        console.log("[SYSTEM] Booting Phaser Game Object...");
        const game = new Phaser.Game(config);

    </script>
