def generate_game():
    # Validate session fingerprint.
    fingerprint = session.get('fingerprint')
    if not fingerprint:
        current_app.logger.warning("No fingerprint found.")
        return jsonify({"error": "Identity missing. Refresh page."}), 400
    data = request.json
    description = data.get('description', '').strip()
    # Log description metrics.
    current_app.logger.info(f"Desc length: {len(description)}")
    if not description:
        current_app.logger.warning("Missing description.")
        return jsonify({"error": "Description required."}), 400
    # Enforce description limit.
    if len(description) > 2000:
        current_app.logger.warning("Description exceeded length limit.")
        return jsonify({"error": "Description too long (max 2000 chars)."}), 400

        cred_req = requests.post(
            f"{LOGINMANAGER_BASE_URL}/get_credits",
            json={"domain": request.host.split(":")[0], "fingerprint": fingerprint, "email": session.get("email", "")},

    except Exception as e:

        return jsonify({"error": "Failed to verify balance."}), 502

import logging
import time
from flask import Flask, render_template, send_from_directory
# Setup basic logging.
logging.basicConfig(level=logging.DEBUG)
def create_app():
    # Import blueprints.
    from _Game import game_bp
    from _Billing_Routes import billing_bp
    from _Auth import auth_bp
    from _Blog_Functions import Blog_Functions
    app = Flask(__name__)
    # Secure application keys.
    app.secret_key = os.environ['APP_SEC']
    app.config["SHOP_KEY"] = os.environ.get('SHOP_KEY')
    # Register all blueprints.
    app.register_blueprint(auth_bp, url_prefix='/auth')
    # FIX: Removed url_prefix so the external gateway can hit the /final route at the root level.
    app.register_blueprint(billing_bp)
    app.register_blueprint(game_bp)
    @app.route('/favicon.ico')
    def favicon():
        # Log favicon request.
        app.logger.info("Route hit: /favicon.ico")
        return send_from_directory(
            os.path.join(app.root_path, 'static', 'icons'),
            'favicon.ico',
            mimetype='image/vnd.microsoft.icon'
        )
    @app.route('/blog')
    def blog():
        # Log blog request.
        app.logger.info("Route hit: /blog")
        # Initialize blog service.
        blog_functions = Blog_Functions()
        blog_functions.create_blog_entry()
        # Fetch data.
        blog_data = blog_functions.get_blog_data()
        # Format timestamps.
        for entry in blog_data:
            entry['timestamp'] = time.strftime('%Y-%m-%d %H:%M', time.localtime(entry.get('unix_timestamp', 0)))
            entry['formatted_hashtags'] = ' '.join(entry.get('hashtags', []))
        return render_template('blog.html', blog_data=blog_data)
    return app
app = create_app()
if __name__ == '__main__':
    # Run development server.
    app.run(debug=True)

Here is the extracted information from the provided document regarding how to track the progress, status, and duration of requests from Anthropic models:
**Message Batches API Progress Tracking**
For asynchronous batch processing, you can track the progress of your requests by polling the API to see how many individual requests have completed.
*
**Processing Status**: The batch response includes a `processing_status` field that indicates the current stage of the batch, starting as `"in_progress"` and updating to `"ended"` once all requests finish.
*
**Request Counts**: You can calculate completion progress using the `request_counts` object. It breaks down the batch into the number of requests that are `"processing"`, `"succeeded"`, `"errored"`, `"canceled"`, or `"expired"`.
*
**Timestamps**: The API returns `created_at`, `ended_at`, and `expires_at` timestamps (note that batches expire after 24 hours) which can help track how long the batch has been running.
**Real-Time Streaming**
For synchronous requests, while the API does not provide an upfront time estimate, you can provide real-time progress to the user by enabling streaming (`"stream": true`).
* Instead of waiting for the full response to finish, the API pushes server-sent events (SSE) to the client to keep the connection alive.
* You receive `content_block_delta` events (which stream text or tool JSON sequentially) to show live generation progress as the model works.
**Token Usage Budgets**
For long-running tasks, Claude 4.5 models feature context awareness where the API updates the remaining token capacity after each tool call.
* The API returns a system warning showing exact usage metrics (e.g., `<system_warning>Token usage: 35000/200000; 165000 remaining</system_warning>`).
* This acts as a "clock" indicating how much working capacity is left before the workflow must be summarized or concluded.
**Execution Duration Metrics**
The SDK result messages and task outputs return exact execution times via `duration_ms` and `duration_api_ms` fields.
While these are returned post-execution, you can log them to calculate historical average completion times for specific tasks to provide better estimates to users in the future.

def track_batch_progress(batch_id: str):
    # Initialize the client
    client = anthropic.Anthropic()
    logging.debug(f"Starting poll for batch: {batch_id}")
    while True:
        # Fetch current batch state
        batch = client.messages.batches.retrieve(batch_id)
        status = batch.processing_status
        logging.info(f"Current batch status: {status}")
        if status == "ended":
            # Exit loop on completion
            logging.debug("Batch processing has finished.")
            break
        # Log exact completion metrics
        counts = batch.request_counts
        logging.debug(f"Progress counts: {counts}")
        # Pause before polling again
        logging.info("Sleeping 10s before next poll.")
        time.sleep(10)

import os
import json
import logging
from _Utils import acquire_lock, release_lock
# Setup basic logging configuration.
logging.basicConfig(level=logging.DEBUG)
# Define local storage path.
STORAGE_FILE = "generation_times.json"
LOCK_FILE = STORAGE_FILE + ".lock"
MAX_SAMPLES = 50
def load_times():
    # Check if storage file exists.
    logging.debug(f"Checking for file: {STORAGE_FILE}")
    if not os.path.exists(STORAGE_FILE):
        # Log missing file.
        logging.info("Storage file not found.")
        return []
    try:
        # Read and parse JSON file.
        logging.debug("Opening file for reading.")
        with open(STORAGE_FILE, 'r') as f:
            data = json.load(f)
        # Log successful read.
        logging.info(f"Loaded {len(data)} time records.")
        return data
    except Exception as e:
        # Log file read error.
        logging.error(f"Error reading file: {e}")
        return []
def save_time(duration_ms):
    # Log save request.
    logging.debug(f"Saving new duration: {duration_ms}ms")
    # Acquire cross-process lock.
    if acquire_lock(LOCK_FILE):
        try:
            # Load existing time records.
            times = load_times()
            # Append new duration.
            times.append(duration_ms)
            # Cap array size to prevent bloat.
            times = times[-MAX_SAMPLES:]
            logging.debug("Appended duration to list.")
            try:
                # Write list to JSON file.
                logging.debug("Opening file for writing.")
                with open(STORAGE_FILE, 'w') as f:
                    json.dump(times, f)
                # Log successful write.
                logging.info("Successfully saved times to disk.")
            except Exception as e:
                # Log write error.
                logging.error(f"Error writing file: {e}")
        finally:
            # Ensure lock is released.
            release_lock(LOCK_FILE)
    else:
        logging.error("Failed to acquire lock for saving.")
def get_average_time():
    # Log average calculation request.
    logging.debug("Calculating average generation time.")
    # Load current times.
    times = load_times()
    if not times:
        # Return default if no data.
        logging.info("No data. Returning default 15000ms.")
        return 15000.0
    # Calculate mathematical average.
    avg = sum(times) / len(times)
    # Log calculated average.
    logging.info(f"Calculated average: {avg:.2f}ms")
    return avg
if __name__ == "__main__":
    # Log script start.
    logging.debug("Progress tracker execution started.")
    # Test saving a time.

    save_time(14500)
    # Test retrieving average.
    logging.info("Testing average calculation.")
    avg_time = get_average_time()
    # Log script end.
    logging.debug("Progress tracker execution finished.")

            constructor(scene, x, y, key, isPlayer1) {
                this.scene = scene;
                this.isP1 = isPlayer1;
                this.key = key;
                // Stats
                this.maxHealth = 1000;
                this.health = this.maxHealth;

                this.sprite.setBounce(0);

                this.hitbox = scene.physics.add.sprite(x, y, 'hitbox_tex');
                this.hitbox.setVisible(false);
                this.hitbox.body.setAllowGravity(false);
                // Opponent Reference (set later)
                this.opponent = null;
                LOG(`Fighter ${this.key} initialized at (${x}, ${y}).`);
            }
            setOpponent(opponentFighter) {
                this.opponent = opponentFighter;
            }
            update(inputs) {
                if (this.currentState === STATES.KO) return;
                const onGround = this.sprite.body.touching.down;
                this.faceOpponent();
                this.updateHitboxPosition();
                // State Machine Switch
                switch (this.currentState) {
                    case STATES.IDLE:
                    case STATES.WALK:
                    case STATES.CROUCH:
                        this.handleNeutralInput(inputs, onGround);
                        break;
                    case STATES.JUMP:
                        this.handleAirInput(inputs, onGround);
                        break;
                    case STATES.ATTACK:
                        this.processAttackFrameData();
                        break;
                    case STATES.HITSTUN:
                    case STATES.BLOCKSTUN:
                        this.processStun();
                        break;
                }

            handleNeutralInput(inputs, onGround) {
                if (!onGround) {
                    this.changeState(STATES.JUMP);
                    return;
                }
                // Attacks take priority
                if (inputs.lightPunch && Phaser.Input.Keyboard.JustDown(inputs.lightPunch)) {

                    return;
                }
                if (inputs.heavyPunch && Phaser.Input.Keyboard.JustDown(inputs.heavyPunch)) {
                    this.executeAttack({ damage: 120, startup: 8, active: 4, recovery: 20, hitstun: 25, blockstun: 15, pushback: 400, isHigh: true });
                    return;
                }
                // Movement & Guard
                let isMoving = false;
                let isCrouching = inputs.down.isDown;
                if (isCrouching) {
                    this.sprite.setVelocityX(0);
                    this.sprite.setScale(1, 0.7); // Visual crouch
                    this.changeState(STATES.CROUCH);
                } else {
                    this.sprite.setScale(1, 1);
                    if (inputs.left.isDown) {
                        this.sprite.setVelocityX(-this.walkSpeed);
                        isMoving = true;
                    } else if (inputs.right.isDown) {
                        this.sprite.setVelocityX(this.walkSpeed);
                        isMoving = true;
                    }
                    if (inputs.up.isDown) {
                        this.sprite.setVelocityY(this.jumpVelocity);
                        this.changeState(STATES.JUMP);
                        LOG(`${this.key} Jumped.`);
                    } else {
                        this.changeState(isMoving ? STATES.WALK : STATES.IDLE);
                    }
                }
            }
            handleAirInput(inputs, onGround) {
                if (onGround) {
                    this.changeState(STATES.IDLE);
                    this.sprite.setVelocityX(0); // Stop momentum on land
                }
            }
            executeAttack(attackData) {
                this.changeState(STATES.ATTACK);
                this.currentAttack = { ...attackData, timer: 0, hasHit: false };
                this.sprite.setVelocityX(0);
                this.sprite.setTint(0xffaa00); // Visual indicator of attack
                LOG(`${this.key} executed attack. Startup: ${attackData.startup}f`);
            }
            processAttackFrameData() {
                if (!this.currentAttack) return;
                const atk = this.currentAttack;
                atk.timer++;
                // Startup Phase
                if (atk.timer < atk.startup) {
                    // Waiting for active frames
                }
                // Active Phase
                else if (atk.timer >= atk.startup && atk.timer < (atk.startup + atk.active)) {
                    this.hitbox.setVisible(true);
                    // Collision Check
                    if (!atk.hasHit && this.scene.physics.overlap(this.hitbox, this.opponent.sprite)) {
                        atk.hasHit = true;
                        this.opponent.receiveHit(atk, this);
                    }
                }
                // Recovery Phase
                else if (atk.timer >= (atk.startup + atk.active) && atk.timer < (atk.startup + atk.active + atk.recovery)) {
                    this.hitbox.setVisible(false);
                    this.sprite.clearTint();
                }
                // End Attack
                else {
                    this.hitbox.setVisible(false);
                    this.sprite.clearTint();

                    this.changeState(STATES.IDLE);
                }

                    this.currentState = newState;
                    this.stateTimer = 0;
                }

        // ENTITY COMPONENT SYSTEM

        class Entity {
            constructor(scene, x, y, z, model) {
                this.scene = scene;
                this.id = Math.random().toString(36).substr(2, 9);
                this.pos = new Vector3(x, y, z);
                this.rot = new Vector3(0, 0, 0);
                this.vel = new Vector3(0, 0, 0);
                this.model = model;
                this.active = true;
                this.scale = 1.0;
                this.type = 'Generic';
                Logger.info('ENTITY_LIFECYCLE', 'Spawned', { type: this.type, id: this.id, x, y, z });
            }
            update(dt) {
                this.pos.add(new Vector3(this.vel.x * dt, this.vel.y * dt, this.vel.z * dt));
            }
            destroy() {
                this.active = false;
                Logger.info('ENTITY_LIFECYCLE', 'Destroyed', { type: this.type, id: this.id });
            }
        }
        class Player extends Entity {
            constructor(scene, x, y, z) {
                super(scene, x, y, z, Models.Arwing);
                this.type = 'Player';
                this.shield = 100;
                this.banking = 0; // -1 to 1
                this.pitching = 0;
            }
            update(dt) {
                super.update(dt);
                // Apply banking visual rotation
                this.rot.z = Phaser.Math.Linear(this.rot.z, this.banking * 0.8, 0.1);
                this.rot.x = Phaser.Math.Linear(this.rot.x, this.pitching * 0.4, 0.1);
                // Boundaries
                const limitX = 400; const limitY = 300;
                if (this.pos.x > limitX) this.pos.x = limitX;
                if (this.pos.x < -limitX) this.pos.x = -limitX;
                if (this.pos.y > limitY) this.pos.y = limitY;
                if (this.pos.y < -limitY) this.pos.y = -limitY;
            }
            takeDamage(amt) {
                this.shield -= amt;
                Logger.warn('COMBAT', 'Player Damaged', { amount: amt, remainingShield: this.shield });
                document.getElementById('hud-shield').innerText = `SHIELD: ${'|'.repeat(Math.max(0, Math.ceil(this.shield/10)))}`;
                if (this.shield <= 0) {
                    Logger.error('GAME_STATE', 'Player Destroyed', {});
                    this.destroy();
                }
            }
        }
        class Projectile extends Entity {
            constructor(scene, x, y, z, velZ, isEnemy = false) {
                super(scene, x, y, z, Models.Laser);
                this.type = 'Projectile';
                this.vel.z = velZ;
                this.isEnemy = isEnemy;
                this.life = 100; // frames
            }
            update(dt) {
                super.update(dt);
                this.life--;
                if (this.life <= 0 || this.pos.z > 5000 || this.pos.z < -1000) this.destroy();
            }

        class Enemy extends Entity {
            constructor(scene, x, y, z) {
                super(scene, x, y, z, Models.EnemyDrone);
                this.type = 'Enemy';
                this.hp = 20;
                this.vel.z = -800; // Move towards player
                this.sinOffset = Math.random() * Math.PI * 2;

                super.update(dt);
                this.rot.x += 0.05;
                this.rot.y += 0.05;
                this.pos.x += Math.sin(this.scene.time.now / 500 + this.sinOffset) * 5;
                if (this.pos.z < -500) this.destroy(); // Passed camera

        // MAIN PHASER SCENE

        class MainScene extends Phaser.Scene {
            constructor() {
                super({ key: 'MainScene' });
                this.entities = [];
                this.wireframeMode = false;
                this.fov = 250;
                this.camDist = 250;
                this.score = 0;
                this.gameTime = 0;
                this.keyMap = {};
            }
            create() {
                Logger.sys('SCENE', 'MainScene Created');
                window.gameEngine = this;
                this.graphics = this.add.graphics();
                // Setup Player
                this.player = new Player(this, 0, 0, -100);
                this.entities.push(this.player);
                // Setup Inputs
                this.cursors = this.input.keyboard.createCursorKeys();
                this.fireKey = this.input.keyboard.addKey(Phaser.Input.Keyboard.KeyCodes.SPACE);
                this.bombKey = this.input.keyboard.addKey(Phaser.Input.Keyboard.KeyCodes.B);
                // Input binding for robust logging
                this.setupInputLogging();
                // Spawners
                this.time.addEvent({ delay: 2000, callback: this.spawnWave, callbackScope: this, loop: true });
                // Background Stars
                this.stars = Array.from({length: 300}, () => new Vector3(
                    Phaser.Math.Between(-2000, 2000),
                    Phaser.Math.Between(-2000, 2000),
                    Phaser.Math.Between(0, 4000)
                ));
            }
            setupInputLogging() {
                const keysToTrack = ['UP', 'DOWN', 'LEFT', 'RIGHT', 'SPACE', 'B'];
                keysToTrack.forEach(k => {
                    this.input.keyboard.on(`keydown-${k}`, () => {
                        if (!this.keyMap[k]) {
                            this.keyMap[k] = true;
                            Logger.info('INPUT', 'Key Down', { key: k });
                        }
                    });
                    this.input.keyboard.on(`keyup-${k}`, () => {
                        this.keyMap[k] = false;
                        Logger.info('INPUT', 'Key Up', { key: k });
                    });
                });
            }
            toggleWireframe() {
                this.wireframeMode = !this.wireframeMode;
                Logger.sys('RENDERER', 'Wireframe Mode Toggled', { state: this.wireframeMode });
            }
            setFOV(val) { this.fov = val; Logger.sys('RENDERER', 'FOV Changed', { value: val }); }
            setCamDist(val) { this.camDist = val; Logger.sys('RENDERER', 'Camera Z Changed', { value: val }); }
            addScore(amt) {
                this.score += amt;
                document.getElementById('hud-score').innerText = `SCORE: ${this.score.toString().padStart(6, '0')}`;
            }
            spawnWave() {
                if (!this.player.active) return;
                const count = Phaser.Math.Between(1, 3);
                Logger.sys('SPAWNER', 'Spawning Wave', { count });
                for(let i=0; i<count; i++) {
                    const e = new Enemy(
                        this,
                        Phaser.Math.Between(-600, 600),
                        Phaser.Math.Between(-400, 400),
                        3000 + (i * 200)
                    );
                    this.entities.push(e);
                }
            }
            fireLaser() {
                if (!this.player.active) return;
                Logger.info('COMBAT', 'Player Fired Laser');
                // Twin lasers
                const l1 = new Projectile(this, this.player.pos.x - 20, this.player.pos.y, this.player.pos.z + 40, 2000, false);
                const l2 = new Projectile(this, this.player.pos.x + 20, this.player.pos.y, this.player.pos.z + 40, 2000, false);
                this.entities.push(l1, l2);
            }
            handleInput(dt) {
                if (!this.player.active) return;
                const speed = 600 * dt;
                this.player.banking = 0;
                this.player.pitching = 0;
                if (this.cursors.left.isDown) { this.player.pos.x -= speed; this.player.banking = 1; }
                if (this.cursors.right.isDown) { this.player.pos.x += speed; this.player.banking = -1; }
                if (this.cursors.up.isDown) { this.player.pos.y -= speed; this.player.pitching = -1; }
                if (this.cursors.down.isDown) { this.player.pos.y += speed; this.player.pitching = 1; }
                if (Phaser.Input.Keyboard.JustDown(this.fireKey)) {
                    this.fireLaser();
                }
            }
            checkCollisions() {
                const enemies = this.entities.filter(e => e.type === 'Enemy');
                const lasers = this.entities.filter(e => e.type === 'Projectile' && !e.isEnemy);
                // Laser vs Enemy O(N*M) - fine for small counts
                for (let l of lasers) {
                    if (!l.active) continue;
                    for (let e of enemies) {
                        if (!e.active) continue;
                        // Simple 3D AABB/Sphereish collision
                        const dx = l.pos.x - e.pos.x;
                        const dy = l.pos.y - e.pos.y;
                        const dz = l.pos.z - e.pos.z;
                        const distSq = dx*dx + dy*dy + dz*dz;
                        if (distSq < 4000) { // roughly 60 units
                            l.destroy();
                            e.hit(10);
                            break;
                        }
                    }
                }
                // Player vs Enemy
                if (this.player.active) {
                    for (let e of enemies) {
                        if (!e.active) continue;
                        const dx = this.player.pos.x - e.pos.x;
                        const dy = this.player.pos.y - e.pos.y;
                        const dz = this.player.pos.z - e.pos.z;
                        if (dx*dx + dy*dy + dz*dz < 10000) { // roughly 100 units
                            this.player.takeDamage(20);
                            e.destroy();
                        }
                    }
                }
            }
            update(time, delta) {
                const dt = delta / 1000;
                this.gameTime += dt;
                document.getElementById('hud-time').innerText = `TIME: ${Math.floor(this.gameTime/60).toString().padStart(2,'0')}:${Math.floor(this.gameTime%60).toString().padStart(2,'0')}`;
                this.graphics.clear();
                this.handleInput(dt);
                this.checkCollisions();
                // Update entities
                for (let i = this.entities.length - 1; i >= 0; i--) {
                    let e = this.entities[i];
                    e.update(dt);
                    if (!e.active) {
                        this.entities.splice(i, 1);
                    }
                }
                const cx = this.cameras.main.width / 2;
                const cy = this.cameras.main.height / 2;
                // Render Stars
                this.graphics.fillStyle(0xffffff, 0.8);
                for (let star of this.stars) {
                    star.z -= 1000 * dt;
                    if (star.z < 0) {
                        star.z = 4000;
                        star.x = Phaser.Math.Between(-2000, 2000);
                        star.y = Phaser.Math.Between(-2000, 2000);
                    }
                    const proj = Math3D.project(star, this.fov, this.camDist, cx*2, cy*2);
                    if (proj.visible) {
                        this.graphics.fillRect(proj.x, proj.y, proj.scale * 3, proj.scale * 3);
                    }
                }
                // Collect and sort polygons for drawing (Painter's Algorithm)
                let renderQueue = [];
                for (let e of this.entities) {
                    if (!e.model) continue;
                    // Transform vertices into world space
                    const worldVerts = e.model.vertices.map(v =>
                        Math3D.transform(v, e.rot.x, e.rot.y, e.rot.z, e.pos.x, e.pos.y, e.pos.z)
                    );
                    for (let face of e.model.faces) {
                        const pts = face.indices.map(idx => worldVerts[idx]);
                        // Backface culling rough estimation (using first 3 points)
                        if (pts.length >= 3) {
                            const v1 = pts[1].clone().sub(pts[0]);
                            const v2 = pts[2].clone().sub(pts[0]);
                            // Cross product Z
                            const normalZ = v1.x * v2.y - v1.y * v2.x;
                            // Only strictly apply culling if we aren't heavily rotating, simplified for retro feel
                            // For true 3D, we need vector to camera. Let's do a simple depth sort first.
                        }
                        // Project
                        let valid = true;
                        const projPts = pts.map(p => {
                            const pr = Math3D.project(p, this.fov, this.camDist, cx*2, cy*2);
                            if (!pr.visible) valid = false;
                            return pr;
                        });
                        if (valid) {
                            // Calculate average Z for sorting
                            const avgZ = pts.reduce((sum, p) => sum + p.z, 0) / pts.length;
                            renderQueue.push({ points: projPts, color: face.color, z: avgZ });
                        }
                    }
                }
                // Sort queue furthest to closest
                renderQueue.sort((a, b) => b.z - a.z);
                // Draw Queue
                for (let poly of renderQueue) {
                    this.graphics.lineStyle(1, 0x000000, 1); // Dark outline for retro flat look
                    if (!this.wireframeMode) {
                        this.graphics.fillStyle(poly.color, 1.0);
                        this.graphics.beginPath();
                        this.graphics.moveTo(poly.points[0].x, poly.points[0].y);
                        for(let i=1; i<poly.points.length; i++) {
                            this.graphics.lineTo(poly.points[i].x, poly.points[i].y);
                        }
                        this.graphics.closePath();
                        this.graphics.fillPath();
                        this.graphics.strokePath();
                    } else {
                        // Wireframe
                        this.graphics.lineStyle(2, poly.color, 1.0);
                        this.graphics.beginPath();
                        this.graphics.moveTo(poly.points[0].x, poly.points[0].y);
                        for(let i=1; i<poly.points.length; i++) {
                            this.graphics.lineTo(poly.points[i].x, poly.points[i].y);
                        }
                        this.graphics.closePath();
                        this.graphics.strokePath();
                    }
                }
                // Horizon line pseudo-effect
                if (this.wireframeMode) {
                    this.graphics.lineStyle(1, 0x00ffcc, 0.3);
                    const horizonY = cy + (this.player.pitching * 50);
                    this.graphics.strokeLineShape(new Phaser.Geom.Line(0, horizonY, cx*2, horizonY));
                }
            }
        }
        // ==========================================
        // ENGINE BOOTSTRAP
        // ==========================================
        const config = {
            type: Phaser.AUTO,
            width: 1280,
            height: 720,
            parent: 'game-wrapper',
            scene: [MainScene],
            backgroundColor: '#000000',
            fps: { target: 60, forceSetTimeOut: true },
            antialias: false // Crisp retro edges
        };
        const game = new Phaser.Game(config);
        Logger.sys('BOOT', 'Phaser Engine Started', { resolution: '1280x720', antialias: false });
    </script>
</body>

    const FLASK_API_URL = "/generate_game";
    let isGameView = false;
    const BACKEND_AVG_TIME = {{ avg_time|default(15000)|int }};
    const toggleBtn = document.getElementById('toggle-btn');
    const promptArea = document.getElementById('prompt-area');
    const gameArea = document.getElementById('game-area');
    const iframe = document.getElementById('game-frame');
    const createBtn = document.getElementById('create-btn');
    const progressBar = document.getElementById('progress-bar');
    const loadingOverlay = document.getElementById('loading');
    const descInputBox = document.getElementById('game-desc');
    // Add explicit Enter key handling for the textarea to trigger the button
    descInputBox.addEventListener('keydown', (e) => {
        if (e.key === 'Enter' && !e.shiftKey) {
            console.log("UI Event: Enter pressed in description box - triggering create button");
            e.preventDefault();
            createBtn.click();
        }
    });
    // Force focus into game iframe.
    function forceGameFocus() {
        if (!isGameView || !iframe) return;
        console.log("UI Event: Forcing focus into iframe.");
        iframe.focus();
        if (iframe.contentWindow) {
            iframe.contentWindow.focus();
            iframe.contentWindow.postMessage({ type: 'engine_focus' }, '*');
        }
    }
    // Global parent click logging.
    document.addEventListener('click', (e) => {
        console.log(`UI Event: Parent click. Tag: ${e.target.tagName}, ID: ${e.target.id}`);
    });
    // Track text input interactions.
    descInputBox.addEventListener('input', () => {
        console.log(`UI Event: Description input changed. Length: ${descInputBox.value.length}`);
    });
    // Transfer keyboard focus to game.
    window.addEventListener('keydown', (e) => {
        console.log(`UI Event: Parent received keydown: ${e.key}`);
        // CRITICAL FIX: Don't intercept if user is typing in textarea or clicking button
        if (document.activeElement === descInputBox ||
            document.activeElement === createBtn ||
            !isGameView) {
            console.log(`Input Bridge: Allowing native handling for ${e.key}`);
            return; // Let the browser handle it normally
        }
        if (isGameView && document.activeElement !== iframe) {
            console.log(`Input Bridge: Transferring focus to game without synthetic events.`);
            forceGameFocus();
        }
    });
    // Update UI view states.
    function updateViewState() {
        console.log(`UI Event: View state -> Game:${isGameView}`);
        if (isGameView) {
            toggleBtn.style.gridColumn = '1 / 2';
            toggleBtn.innerHTML = '&gt;';
            promptArea.style.display = 'none';
            gameArea.style.display = 'flex';
            gameArea.style.gridColumn = '2 / 33';
            gameArea.style.gridRow = '1';
            loadingOverlay.style.display = 'none';
            requestAnimationFrame(() => {
                setTimeout(forceGameFocus, 10);
                setTimeout(forceGameFocus, 300);
            });
        } else {
            toggleBtn.style.gridColumn = '32 / 33';
            toggleBtn.innerHTML = '&lt;';
            promptArea.style.display = 'flex';
            promptArea.style.gridColumn = '1 / 32';
            promptArea.style.gridRow = '1';
            gameArea.style.display = 'none';
        }
    }
    // Initial layout calculation.
    console.log("UI Event: Initial view state computation.");
    updateViewState();
    toggleBtn.addEventListener('click', () => {
        console.log("UI Event: View toggle button clicked.");
        isGameView = !isGameView;
        updateViewState();
    });
    iframe.onload = () => {
        console.log("UI Event: Iframe loaded fully.");
    };
    createBtn.addEventListener('click', async () => {
        const descInput = descInputBox.value.trim();
        console.log(`UI Event: Create Game clicked. Input length: ${descInput.length}`);
        if (!descInput) {
            console.warn("Validation Error: Description box is empty.");
            alert("Describe the game first, captain!");
            return;
        }
        createBtn.disabled = true;
        createBtn.innerText = "Generating...";
        loadingOverlay.style.display = 'flex';
        iframe.srcdoc = "";
        progressBar.style.width = '0%';
        console.log("UI Event: Showing loading overlay.");
        let startTime = Date.now();
        let progressInterval = setInterval(() => {
            let elapsed = Date.now() - startTime;
            let progress = Math.min(98, (elapsed / BACKEND_AVG_TIME) * 100);
            progressBar.style.width = `${progress}%`;
            if (progress > 80) {
                document.getElementById('status-update').innerText = "Almost there... finalizing code.";
            }
        }, 200);
        try {
            console.log("Network Event: Fetching generated game from AI backend.");
            const response = await fetch(FLASK_API_URL, {
                method: 'POST',
                credentials: 'same-origin', // Send session cookies securely.
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ description: descInput })
            });
            const data = await response.json();
            if (!response.ok) {
                console.error(`Network Error: ${response.status} - ${data.error}`);
                if (response.status === 401) window.location.href = "{{ url_for('auth.login') }}";
                if (response.status === 403) window.location.href = "{{ url_for('billing.tiers') }}";
                throw new Error(data.error || "Generation failed.");
            }
            console.log("Network Event: Game generated successfully.");
            clearInterval(progressInterval);
            progressBar.style.width = '100%';
            // Update top bar credits ID match.
            const navCredits = document.getElementById('credit-amount');
            if (navCredits && data.credits_remaining !== undefined) {
                console.log(`UI Event: Updating credits to ${data.credits_remaining}.`);
                navCredits.innerText = data.credits_remaining + " Credits";
            }
            console.log("UI State: Injecting raw HTML into iframe.");
            iframe.srcdoc = data.game_html;
            setTimeout(() => {
                console.log("UI Event: Transitioning to game view.");
                isGameView = true;
                updateViewState();
            }, 50);
        } catch (error) {
            console.error("System Error: Game generation caught exception.", error);
            clearInterval(progressInterval);
            alert(error.message);
            loadingOverlay.style.display = 'none';
        } finally {
            console.log("UI Event: Resetting create button state.");
            createBtn.disabled = false;
            createBtn.innerText = "Create Game";
        }
    });
    gameArea.addEventListener('mousedown', () => {
        console.log("UI Event: Game area padding clicked.");
        forceGameFocus();
    });
</script>
